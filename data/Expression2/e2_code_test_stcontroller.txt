@name Test Control
@inputs But RF
@outputs PV CN TT EE RR
@persist T:fsensor E:entity C:stcontroller V2:vector2
@trigger 
@model models/props_phx/construct/metal_wire1x1x1.mdl

runOnTick(1)

if(first() || dupefinished())
{
    V2 = vec2(1.1,0.5)   
    E = entity()
    T = E:setFSensor(E:massCenterL(),vec(0,0,1), -225):addClassHit("prop_physics")
    C = newStController():setGain(1700, 6800, 7800):setWindup(-9000,9000):setPowerPI(V2):dumpConsole("TEST")
    E:setMass(250)
}

PV = T:smpLocal():getFractionLength()
CN = C:isActive(But):setState(RF, PV):getControl()


TT = C:getTimeDelta()

E:applyForce(CN * vec(0,0,1))

EE = C:getErrorNow()
RR = C:getTimeRatio()
BB = C:getTimeBench()
